    –û—Ç–≤–µ—á–∞–µ—Ç –∑–∞ –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ –≥—Ä–∞—Ñ–∏–∫–∏, —Ä–∞–±–æ—Ç—É —Å —Ü–≤–µ—Ç–∞–º–∏ –∏ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞–º–∏.

    –ê—Ç—Ä–∏–±—É—Ç—ã:
        res_x (int): –®–∏—Ä–∏–Ω–∞ –æ–∫–Ω–∞.
        res_y (int): –í—ã—Å–æ—Ç–∞ –æ–∫–Ω–∞.
        rf (ReferenceFrame): –û–±—ä–µ–∫—Ç, –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—é—â–∏–π —Å–∏—Å—Ç–µ–º—É –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç.
        screen (pygame.Surface): –û–±—ä–µ–∫—Ç —ç–∫—Ä–∞–Ω–∞ Pygame, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–º –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç —Ä–∏—Å–æ–≤–∞–Ω–∏–µ.
        __color (tuple): –¶–≤–µ—Ç, –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã–π –¥–ª—è —Ä–∏—Å–æ–≤–∞–Ω–∏—è –æ–±—ä–µ–∫—Ç–æ–≤.
    """

    def __init__(self, res_x, res_y, rf: ReferenceFrame):
        """
        –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –æ–±—ä–µ–∫—Ç–∞ Drawer.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            res_x (int): –®–∏—Ä–∏–Ω–∞ —ç–∫—Ä–∞–Ω–∞.
            res_y (int): –í—ã—Å–æ—Ç–∞ —ç–∫—Ä–∞–Ω–∞.
            rf (ReferenceFrame): –°–∏—Å—Ç–µ–º–∞ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –¥–ª—è —Ä–∏—Å–æ–≤–∞–Ω–∏—è.
        """
        self.res_x = res_x
        self.res_y = res_y
        self.rf = rf
        self.screen = None
        self.__color = (255, 255, 255)  # –¶–≤–µ—Ç –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é (–±–µ–ª—ã–π)

    def initialize(self, caption: str):
        """
        –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è Pygame, —Å–æ–∑–¥–∞–Ω–∏–µ –æ–∫–Ω–∞ –∏ –∑–∞–ø–æ–ª–Ω–µ–Ω–∏–µ —ç–∫—Ä–∞–Ω–∞ —Ñ–æ–Ω–æ–º.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            caption (str): –ó–∞–≥–æ–ª–æ–≤–æ–∫ –æ–∫–Ω–∞.
        """
        pygame.init()
        self.screen = pygame.display.set_mode((self.res_x, self.res_y))
        pygame.display.set_caption(caption)
        self.screen.fill((255, 255, 255))

    @property
    def color(self):
        """
        –ì–µ—Ç—Ç–µ—Ä –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ —Ü–≤–µ—Ç–∞.

        –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
            tuple: –¶–≤–µ—Ç –≤ —Ñ–æ—Ä–º–∞—Ç–µ (r, g, b).
        """
        return self.__color

    @color.setter
    def color(self, color: tuple):
        """
        –°–µ—Ç—Ç–µ—Ä –¥–ª—è –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ü–≤–µ—Ç–∞.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            color (tuple): –ù–æ–≤—ã–π —Ü–≤–µ—Ç –≤ —Ñ–æ—Ä–º–∞—Ç–µ (r, g, b).
        """
        self.__color = color

    def get_x(self, x: float):
        """
        –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã X –∏–∑ —Å–∏—Å—Ç–µ–º—ã –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –≤ –ø–∏–∫—Å–µ–ª–∏ —ç–∫—Ä–∞–Ω–∞.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            x (float): –ö–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ –ø–æ –æ—Å–∏ X –≤ —Å–∏—Å—Ç–µ–º–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç.

        –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
            int: –ü–∏–∫—Å–µ–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ X –Ω–∞ —ç–∫—Ä–∞–Ω–µ.
        """
        return int(self.rf.origin.x0 + x * self.rf.unit_x.pixels)

    def get_y(self, y: float):
        """
        –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã Y –∏–∑ —Å–∏—Å—Ç–µ–º—ã –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –≤ –ø–∏–∫—Å–µ–ª–∏ —ç–∫—Ä–∞–Ω–∞.
        –ù–∞—á–∞–ª–æ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç (0, 0) –±—É–¥–µ—Ç –Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è –≤ –ª–µ–≤–æ–º –Ω–∏–∂–Ω–µ–º —É–≥–ª—É —ç–∫—Ä–∞–Ω–∞.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            y (float): –ö–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ –ø–æ –æ—Å–∏ Y –≤ —Å–∏—Å—Ç–µ–º–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç.

        –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç:
            int: –ü–∏–∫—Å–µ–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ Y –Ω–∞ —ç–∫—Ä–∞–Ω–µ.
        """
        return int(self.res_y - (self.rf.origin.y0 + y * self.rf.unit_y.pixels))

    def draw_text(self, x: float, y: float, text: str):
        """
        –†–∏—Å–æ–≤–∞–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –Ω–∞ —ç–∫—Ä–∞–Ω–µ –≤ –∑–∞–¥–∞–Ω–Ω–æ–π —Ç–æ—á–∫–µ.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            x (float): –ö–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ X –¥–ª—è —Ç–µ–∫—Å—Ç–∞.
            y (float): –ö–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ Y –¥–ª—è —Ç–µ–∫—Å—Ç–∞.
            text (str): –¢–µ–∫—Å—Ç –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è.
        """
        font = pygame.font.Font(None, 24)  # –®—Ä–∏—Ñ—Ç
        text_surface = font.render(text, True, self.color)  # –°–æ–∑–¥–∞–Ω–∏–µ —Ç–µ–∫—Å—Ç–æ–≤–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        self.screen.blit(text_surface, (self.get_x(x), self.get_y(y)))  # –û—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –Ω–∞ —ç–∫—Ä–∞–Ω–µ

    def draw_line(self, x1: float, y1: float, x2: float, y2: float, width: int = 1):
        """
        –†–∏—Å–æ–≤–∞–Ω–∏–µ –ª–∏–Ω–∏–∏ –Ω–∞ —ç–∫—Ä–∞–Ω–µ –æ—Ç –æ–¥–Ω–æ–π —Ç–æ—á–∫–∏ –¥–æ –¥—Ä—É–≥–æ–π.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            x1 (float): –ù–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ X.
            y1 (float): –ù–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ Y.
            x2 (float): –ö–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ X.
            y2 (float): –ö–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ Y.
            width (int): –®–∏—Ä–∏–Ω–∞ –ª–∏–Ω–∏–∏ (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 1).
        """
        pygame.draw.line(
            self.screen,
            self.color,
            (self.get_x(x1), self.get_y(y1)),
            (self.get_x(x2), self.get_y(y2)),
            width
        )

    def draw_polygon(self, points: np.ndarray, width: int = 1):
        """
        –†–∏—Å–æ–≤–∞–Ω–∏–µ –º–Ω–æ–≥–æ—É–≥–æ–ª—å–Ω–∏–∫–∞ –ø–æ –∑–∞–¥–∞–Ω–Ω—ã–º –≤–µ—Ä—à–∏–Ω–∞–º.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            points (np.ndarray): –ú–∞—Å—Å–∏–≤ —Ç–æ—á–µ–∫ –º–Ω–æ–≥–æ—É–≥–æ–ª—å–Ω–∏–∫–∞ –≤ —Å–∏—Å—Ç–µ–º–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç.
            width (int): –®–∏—Ä–∏–Ω–∞ –ª–∏–Ω–∏–π –º–Ω–æ–≥–æ—É–≥–æ–ª—å–Ω–∏–∫–∞ (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 1).
        """
        pygame.draw.polygon(
            self.screen,
            self.color,
            [(self.get_x(pt[0]), self.get_y(pt[1])) for pt in points],
            width
        )

    def draw_axes(self, x_min, x_max, y_min, y_max):
        """
        –†–∏—Å–æ–≤–∞–Ω–∏–µ –æ—Å–µ–π –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –Ω–∞ —ç–∫—Ä–∞–Ω–µ.
        –û—Å—å X —Ä–∏—Å—É–µ—Ç—Å—è –Ω–∞ —É—Ä–æ–≤–Ω–µ y = 0, –∞ –æ—Å—å Y ‚Äî –Ω–∞ —É—Ä–æ–≤–Ω–µ x = 0.

        –ü–∞—Ä–∞–º–µ—Ç—Ä—ã:
            x_min (float): –ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –æ—Å–∏ X.
            x_max (float): –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –æ—Å–∏ X.
            y_min (float): –ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –æ—Å–∏ Y.
            y_max (float): –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –æ—Å–∏ Y.
        """
        self.draw_line(x_min, 0, x_max, 0, 2)  # –†–∏—Å–æ–≤–∞–Ω–∏–µ X-–æ—Å–∏
        self.draw_line(0, y_min, 0, y_max, 2)  # –†–∏—Å–æ–≤–∞–Ω–∏–µ Y-–æ—Å–∏

    cs = m.cos(angle)

import pygame
import numpy as np
import sys
from graphics import Unit, Origin, ReferenceFrame, Drawer
from transform import translate, scale, rotate

# –ö–æ–Ω—Å—Ç–∞–Ω—Ç—ã
RES_X, RES_Y = 800, 600
SCALE = 0.9  # –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –º–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏—è
ANGLE = np.pi / 32  # –£–≥–æ–ª –ø–æ–≤–æ—Ä–æ—Ç–∞ (œÄ/32)
ITERATIONS = 20  # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–∞—Ü–∏–π

# –ù–∞—á–∞–ª—å–Ω—ã–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –∫–≤–∞–¥—Ä–∞—Ç–∞
square = np.array([
    [2, -2, 1],
    [-2, -2, 1],
    [-2, 2, 1],
    [2, 2, 1]
])

# –ï–¥–∏–Ω–∏—á–Ω—ã–π —Ä–∞–∑–º–µ—Ä –∏ –Ω–∞—á–∞–ª–æ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç
unit = Unit(100)  # –ö–∞–∂–¥–∞—è –µ–¥–∏–Ω–∏—Ü–∞ —Ä–∞–≤–Ω–∞ 100 –ø–∏–∫—Å–µ–ª—è–º
origin = Origin(RES_X / 2, RES_Y / 2)
rf = ReferenceFrame(origin, unit, unit)

# Drawer –¥–ª—è —Ä–∏—Å–æ–≤–∞–Ω–∏—è
drawer = Drawer(RES_X, RES_Y, rf)
drawer.initialize("Square Transformation")

# –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª –ø—Ä–æ–≥—Ä–∞–º–º—ã
running = True
while running:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # –û—á–∏—Å—Ç–∫–∞ —ç–∫—Ä–∞–Ω–∞
    drawer.screen.fill((0, 0, 0))

    # –ù–∞—Ä–∏—Å–æ–≤–∞—Ç—å –æ—Å–∏
    # drawer.color = (255, 255, 255)
    # drawer.draw_axes(-5, 5, -5, 5)

    # –†–∏—Å–æ–≤–∞—Ç—å –≤—Å–µ –∏—Ç–µ—Ä–∞—Ü–∏–∏
    current_square = square.copy()
    for i in range(ITERATIONS + 1):
        # –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ: –º–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏–µ –∏ –ø–æ–≤–æ—Ä–æ—Ç –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–Ω—Ç—Ä–∞
        transform = (
            translate(-origin.x0 / unit.pixels, -origin.y0 / unit.pixels) @  # –°–¥–≤–∏–≥ –∫ —Ü–µ–Ω—Ç—Ä—É
            scale(SCALE, SCALE) @  # –ú–∞—Å—à—Ç–∞–±–∏—Ä–æ–≤–∞–Ω–∏–µ
            rotate(ANGLE) @  # –ü–æ–≤–æ—Ä–æ—Ç
            translate(origin.x0 / unit.pixels, origin.y0 / unit.pixels)  # –í–æ–∑–≤—Ä–∞—Ç –∫ —Ü–µ–Ω—Ç—Ä—É
        )

        # –¶–≤–µ—Ç –∫–≤–∞–¥—Ä–∞—Ç–∞ –∑–∞–≤–∏—Å–∏—Ç –æ—Ç –∏—Ç–µ—Ä–∞—Ü–∏–∏
        drawer.color = (255 - i * 10, 0, i * 12)

        # –ù–∞—Ä–∏—Å–æ–≤–∞—Ç—å –∫–≤–∞–¥—Ä–∞—Ç
        drawer.draw_polygon(current_square)

        # –ü—Ä–∏–º–µ–Ω–∏—Ç—å —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–∞—Ü–∏—é –¥–ª—è —Å–ª–µ–¥—É—é—â–µ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏
        current_square = current_square @ transform.T

    # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ —ç–∫—Ä–∞–Ω–∞
    pygame.display.flip()
    pygame.time.Clock().tick(30)

# –ó–∞–≤–µ—Ä—à–µ–Ω–∏–µ Pygame
pygame.quit()
sys.exit()
